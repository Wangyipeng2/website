public class ChineseFlag extends JPanel
{
   // Set the width and height of the flag
   private static final int FLAG_WIDTH = 800;
   private static final int FLAG_HEIGHT = 480;
   // Main method to set up the window
   public static void main(String[] args)
   {
      JFrame frame = new JFrame("Chinese National Flag");
      ChineseFlag panel = new ChineseFlag();
      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      frame.setSize(FLAG_WIDTH, FLAG_HEIGHT);
      frame.add(panel);

   }
   // Paint method to draw the flag
   @Override
   protected void paintComponent(Graphics g)
   {
      super.paintComponent(g);
      // Get the graphics object
      Graphics2D g2d = (Graphics2D) g;
      // Anti-aliasing for smoother lines
      g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
      // Set the background to red (the flag's background color)
      g2d.setColor(new Color(222, 41, 19));
      // Red color of the flag
      g2d.fillRect(0, 0, FLAG_WIDTH, FLAG_HEIGHT);
      // Draw the large star
      g2d.setColor(Color.YELLOW);
      drawStar(g2d, 100, 80, 50, 20, 5);
      // Draw the smaller stars
      drawSmallStars(g2d);
   }
   // Method to draw a star at a given position
   private void drawStar(Graphics2D g2d, int x, int y, int radius1, int radius2, int numPoints)
   {
      // The angle between each point on the star
      double angle = Math.PI / numPoints;
      // Create a polygon representing the star
      GeneralPath path = new GeneralPath();
      // First point of the star
      path.moveTo(x + Math.cos(0) * radius1, y - Math.sin(0) * radius1);
      // Alternate between outer and inner points
      for (int i = 1; i < 2 * numPoints; i++)
      {
         double currentAngle = i * angle;
         double currentRadius = (i % 2 == 0) ? radius1 : radius2;
         path.lineTo(x + Math.cos(currentAngle) * currentRadius, y - Math.sin(currentAngle) * currentRadius);
      }
      path.closePath();

   {
      int numPoints = 5;
      double outerRadius = radius;
      double innerRadius = radius / 2;
      double angleIncrement = 2 * Math.PI / numPoints;
      
      Path2D path = new Path2D.Double();
      for (int i = 0; i < numPoints; i++)
      {
         double angle = i * angleIncrement;
         double outerX = centerX + outerRadius * Math.cos(angle);
         double outerY = centerY + outerRadius * Math.sin(angle);
         double innerAngle = (i + 0.5) * angleIncrement;
         double innerX = centerX + innerRadius * Math.cos(innerAngle);
         double innerY = centerY + innerRadius * Math.sin(innerAngle);

         else
         {
            path.lineTo(outerX, outerY);
         }
         path.lineTo(innerX, innerY);
      }
      path.closePath();
      g2d.fill(path);

      {

         int n = 0;
         int row = 0;
         int col = 0;
         float sum;
         String grades = "FDCBA";
         String[] headings = new String[9];
         float[][] nums = new float[48][];
         float[] means = new float[9];
         float[] stdvs = new float[9];
         
         while (scanner.hasNextLine())
         {
            String line = scanner.nextLine().trim();
            if (line.isEmpty())
            {
               continue;
            }
            String[] values = line.split("\t");
            if (col == 0)
            {
               nums[row] = new float[9];
            }
            for (int i = 0; i < values.length; i++)
            {
               if (row == 0 || i == 1)
               {
                  if (i == 1)
                  {
                     nums[row][i] = grades.indexOf(values[i]);
                  }
                  else
                  {
                     headings[i] = values[i];
                  }
               }
               else
               {
                  nums[row][i] = Float.parseFloat(values[i]);
               }
            }
            col++;

            {
               col = 0;
               row++;
            }

         {
            sum = 0;
            for (int i = 0; i < n; i++)
            {
               sum += nums[i][j];
            }
            means[j] = sum / n;
         }

         {
            sum = 0;
            for (int i = 0; i < n; i++)
            {
               float diff = nums[i][j] - means[j];
               sum += diff * diff;
            }
            stdvs[j] = (float) Math.sqrt(sum / (n - 1));

public class SimpleHTTPServer
{
   private ServerSocket serverSocket;
   private final int port;
   public SimpleHTTPServer(int port)
   {
      this.port = port;
   }
   public void start()
   {
      try
      {
         serverSocket = new ServerSocket(port);
         System.out.println("Server started on port "+ port);
         while (true)
         {
            Socket clientSocket = serverSocket.accept();
            Thread thread = new Thread(() -> handleClient(clientSocket));
            thread.start();
         }
      }
      catch (IOException e)
      {
         e.printStackTrace();
      }
   }
   private void handleClient(Socket clientSocket)
   {
      try
      {
         BufferedReader in = new BufferedReader(new InputStreamReader(clientSocket.getInputStream()));
         DataOutputStream out = new DataOutputStream(clientSocket.getOutputStream());
         // Read the request headers
         Map headers = readHeaders(in);
         System.out.println(headers);
         // Check if it's a multipart/form-data request
         // Send response
         String response = "HTTP/1.1 200 OK\r\n"
         + "Content-Type: text/plain\r\n"
         + "\r\n"
         + "request received";
         out.writeBytes(response);
         in.close();
         out.close();
         clientSocket.close();
      }
      catch (IOException e)
      {
         e.printStackTrace();
      }
   }
   private Map readHeaders(BufferedReader in) throws IOException
   {
      Map headers = new HashMap<>();
      String line;
      while (!(line = in.readLine()).isEmpty())
      {
         String[] parts = line.split(":", 2);
         if (parts.length == 2)
         {
            headers.put(parts[0].trim(), parts[1].trim());
         }
      }
      return headers;
   }

   {
      SimpleHTTPServer server = new SimpleHTTPServer(8080);
      server.start();

import java.io.DataInputStream;
import java.io.DataOutputStream;
import java.net.ServerSocket;
import java.net.Socket;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;
import java.util.Set;
import java.util.concurrent.Executors;
import java.util.concurrent.ThreadPoolExecutor;

public class GogameServer
{
   Map<String, Set<Integer>> stones = new HashMap<>();
   
   static class Worker extends Thread
   {
      private Socket socket;
      
      public Worker(Socket s)
      {
         socket = s;
      }
      
      private static final int N = 19;
      
      @Override
      public void run()
      {
         try
         {
            DataInputStream dis = new DataInputStream(socket.getInputStream());
            String ip = socket.getInetAddress().toString();
            Set<Integer> coordinates = stones.getOrDefault(ip, new HashSet<>());
            int r = dis.readInt();
            int c = dis.readInt();
            coordinates.add(r * 100 + c);
            
            double distance = 2 * N;
            int minr = 0;
            int minc = 0;
            
            for (int row = 0; row < N; row++)
            {
               for (int col = 0; col < N; col++)
               {
                  if (coordinates.contains(row * 100 + col))
                  {
                     continue;
                  }
                  double dist = Math.sqrt((col - c) * (col - c) + (row - r) * (row - r));
                  if (dist < distance)
                  {
                     distance = dist;
                     minr = row;
                     minc = col;
                  }
               }
            }
            coordinates.add(minr * 100 + minc);
            socket.close();
            try
            {
               Thread.sleep(200);
            }
            catch (InterruptedException e)
            {
               e.printStackTrace();
            }
            sendnum(ip, minr, minc);
         }
         catch (Exception e)
         {
            e.printStackTrace();
         }
      }
   }
   
   private final int port = 2311;
   
   protected void sendnum(String ip, int r, int c)
   {
      new Thread(() ->
      {
         try
         {
            Socket s = new Socket(ip, port);
            DataOutputStream out = new DataOutputStream(s.getOutputStream());
            out.writeInt(r);
            out.writeInt(c);
            s.close();
         }
         catch (Exception e)
         {
            e.printStackTrace();
         }
      }
      ).start();
   }
   
   ThreadPoolExecutor pool = (ThreadPoolExecutor) Executors.newFixedThreadPool(10);
   
   public void receive()
   {
      try
      {
         ServerSocket ss = new ServerSocket(port);
         while (true)
         {
            Socket s = ss.accept();
            Worker w = new Worker(s);
            pool.execute(w);
            w.start();
         }
      }
      catch (Exception e)
      {
         e.printStackTrace();
      }
   }
   
   public static void main(String[] args)
   {
      GogameServer server = new GogameServer();
      server.receive();
   }
}
